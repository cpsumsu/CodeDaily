> Problem: [B141:【異世界コラボ問題】ガーベラ・コレクション](https://paiza.jp/challenges/691/show)

# 思路

> Dp模擬構造題，小心邊界條件即可
>
> 遞堆式為各取來向的方格上的最大數字

$$
mp[i][j][k] += max(mp[i][j - 1][k], mp[i - 1][j][k]);
$$

$$
mp[i][j][k] += max(mp[i][j + 1][k], mp[i + 1][j][k]);
$$

$$
...
$$

$$
...
$$

# 复杂度

时间复杂度:
> $O(nmk)$

空间复杂度:
> $O(nmk)$

int bfs(int k)
{
	int dx = 1, dy = 1;
	int sx = st[k].first, sy = st[k].second;
	int ex = ed[k].first, ey = ed[k].second;
	if (sx < ex && sy < ey)
	{
		for (int i = sx; i <= ex; i += dx)
		{
			for (int j = sy; j <= ey; j += dy)
			{
				if (i == sx && j == sy) continue;
				if (sx == i)
					mp[i][j][k] += mp[i][j - 1][k];
				else if (sy == j)
					mp[i][j][k] += mp[i - 1][j][k];
				else
					mp[i][j][k] += max(mp[i][j - 1][k], mp[i - 1][j][k]);
			}
		}
	}
	else if (sx < ex && sy > ey)
	{
		for (int i = sx; i <= ex; i += dx)
		{
			for (int j = sy; j >= ey; j -= dy)
			{
				if (i == sx && j == sy) continue;
				if (sx == i)
					mp[i][j][k] += mp[i][j + 1][k];
				else if (sy == j)
					mp[i][j][k] += mp[i - 1][j][k];
				else
					mp[i][j][k] += max(mp[i][j + 1][k], mp[i - 1][j][k]);
			}
		}
	}
	else if (sx > ex && sy > ey)
	{
		for (int i = sx; i >= ex; i -= dx)
		{
			for (int j = sy; j >= ey; j -= dy)
			{
				if (i == sx && j == sy) continue;
				if (sx == i)
					mp[i][j][k] += mp[i][j + 1][k];
				else if (sy == j)
					mp[i][j][k] += mp[i + 1][j][k];
				else
					mp[i][j][k] += max(mp[i][j + 1][k], mp[i + 1][j][k]);
			}
		}
	}
	else if (sx > ex && sy < ey)
	{
		for (int i = sx; i >= ex; i -= dx)
		{
			for (int j = sy; j <= ey; j += dy)
			{
				if (i == sx && j == sy) continue;
				if (sx == i)
					mp[i][j][k] += mp[i][j - 1][k];
				else if (sy == j)
					mp[i][j][k] += mp[i + 1][j][k];
				else
					mp[i][j][k] += max(mp[i][j - 1][k], mp[i + 1][j][k]);
			}
		}
	}
	else if (sx == ex)
	{
		int res = 0;
		for (int i = min(sy, ey) + 1; i < max(sy, ey); ++i)
			res += mp[sx][i][k];
		return res;
	}
	else if (sy == ey)
	{
		int res = 0;
		for (int i = min(sx, ex) + 1; i < max(sx, ex); ++i)
			res += mp[i][sy][k];
		return res;
	}
	return mp[ex][ey][k];
}
