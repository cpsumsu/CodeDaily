> 暴力題，甚至不需要差分數組

> $O(tb)$

> $O(1)$

	// 這裡是加速 cin，如果確定代碼沒有問題但又過不了，可以刪掉
	cin.tie(0)->sync_with_stdio(0);

	for (int i = 0; i < t; ++i)
	{
		int a, b;
		cin >> a >> b;
		for (int j = a; j <= b; ++j)
			vis[j]++;
	}
	for (int i = 0; i < 114514; ++i)
	{
		if (!vis[i]) cnt = 1;
		else
		{
			cnt++;
			ans = max(ans, cnt);
		}
	}
	cout << ans - 1 << endl;
	return 0;
}
